<%*
// --- Script Start: Generate Next Branch Topic ---

// 1. Script Initialization and Environment Check
const notice = (msg) => new Notice(msg, 5000);
const log = (msg) => console.log(msg);

let currentFile = app.workspace.getActiveFile();
if (!currentFile) {
    notice("请先打开一个笔记，它将作为新分支主题的参考笔记。");
    return;
}

let parentFolder = currentFile.parent;
let currentFileName = currentFile.name;
let currentFileBaseName = currentFile.basename;

// Helper function for path joining
function joinPath(base, ...parts) {
    let path = base;
    if (path === '/') { path = ''; }
    for (const part of parts) {
        if (part === null || part === undefined || part === "") continue;
        if (path === '' || path.endsWith('/')) { 
            path += part.startsWith('/') ? part.substring(1) : part; 
        } else { 
            path += '/' + (part.startsWith('/') ? part.substring(1) : part); 
        }
    }
    path = path.replace(/(?<!:)\/\//g, '/');
    if (base === '/' && !path.startsWith('/') && path !== '') { path = '/' + path; }
    return path;
}

// 2. Parse Current Filename to get its numerical prefix, or prompt to add one
let currentNumericalPrefix;
const idPatternForExisting = /^(\d+(\.\d+)*)/; // Matches 1, 1.2, 1.2.3, etc.
const idPatternForInput = /^(\d+(\.\d+){1,})$/;   // For validating user input for new ID (must be at least X.Y)

let matchResult = currentFileName.match(idPatternForExisting);

if (matchResult && matchResult[1] && matchResult[1].includes('.')) { // Ensure existing prefix has at least X.Y
    currentNumericalPrefix = matchResult[1];
    log(`Current file has numerical prefix: ${currentNumericalPrefix}`);
} else {
    const noPrefixMessage = matchResult && matchResult[1] ? `当前文件名 "${currentFileName}" 的数字前缀 "${matchResult[1]}" 不符合 "X.Y..." 格式。` : `当前文件名 "${currentFileName}" 没有数字前缀。`;
    log(noPrefixMessage);
    let validInitialInput = false;
    while (!validInitialInput) {
        const userInputInitialNumber = await tp.system.prompt(
            `${noPrefixMessage}\n` +
            `请输入一个有效的初始编号 (格式如 1.2, 1.2.3) 来重命名当前文件，然后才能创建下一个分支主题：`, ""
        );

        if (userInputInitialNumber === null || userInputInitialNumber === undefined) {
            log("User cancelled initial number input for current file.");
            notice("操作已取消：当前文件需要 "X.Y..." 格式的数字前缀。");
            return;
        }

        const initialMatchResult = userInputInitialNumber.trim().match(idPatternForInput);
        if (initialMatchResult && initialMatchResult[1]) {
            const newPrefixForCurrentFile = initialMatchResult[1];
            log(`User provided initial prefix for current file: ${newPrefixForCurrentFile}.`);

            const baseNameAfterStripping = currentFileBaseName.replace(idPatternForExisting, "").trim();
            let renameBase = `${newPrefixForCurrentFile} ${baseNameAfterStripping}`.trim();
            
            let newNameForCurrentFile = renameBase + ".md";
            let newPathForCurrentFile = joinPath(currentFile.parent.path, newNameForCurrentFile);
            
            let renameCounter = 0;
            const originalRenameBase = renameBase;
            while (await app.vault.adapter.exists(newPathForCurrentFile) && newPathForCurrentFile !== currentFile.path) {
                renameCounter++;
                newNameForCurrentFile = `${originalRenameBase}-${renameCounter}.md`;
                newPathForCurrentFile = joinPath(currentFile.parent.path, newNameForCurrentFile);
            }

            try {
                await app.fileManager.renameFile(currentFile, newPathForCurrentFile);
                notice(`当前文件已重命名为: ${newNameForCurrentFile}`);
                
                currentFile = app.vault.getAbstractFileByPath(newPathForCurrentFile);
                if (!currentFile || !(currentFile instanceof tp.obsidian.TFile)) {
                    notice("错误：重命名后无法正确获取文件引用。");
                    log(`Error: Failed to get valid TFile reference after renaming. currentFile is: ${currentFile}`);
                    return;
                }
                
                currentFileName = currentFile.name;
                currentFileBaseName = currentFile.basename;
                parentFolder = currentFile.parent;
                
                currentNumericalPrefix = newPrefixForCurrentFile;
                log(`Successfully renamed current file. New prefix is ${currentNumericalPrefix}`);
                validInitialInput = true;
            } catch (e) {
                notice(`重命名当前文件 "${currentFileName}" (尝试 "${newNameForCurrentFile}") 失败: ${e.message}`);
                log(`Error renaming current file: ${e}`);
                return; 
            }
        } else {
            notice("输入的编号格式不正确，请使用至少两段并用点号分隔的数字 (例如 '1.2' 或 '1.2.3')。");
        }
    }
}
// Now currentNumericalPrefix should be set and have at least two parts (e.g., "X.Y")

// 3. Calculate New File's Base Number String for the Next Branch Topic
const parts = currentNumericalPrefix.split('.');
let newBaseNumberString;

// This logic requires at least two parts (X.Y) in the currentNumericalPrefix
if (parts.length < 2) {
    notice(`错误：处理后的父笔记编号 "${currentNumericalPrefix}" 不足两部分 (X.Y)，无法创建下一个分支主题。`);
    log(`Error: Processed parent prefix "${currentNumericalPrefix}" has fewer than two parts.`);
    return;
}

// Increment the second part (Y)
parts[1] = (parseInt(parts[1], 10) + 1).toString();

// Build the new string: X.(Y+1)
newBaseNumberString = `${parts[0]}.${parts[1]}`;

// For any further parts (Z, A, ... in original X.Y.Z.A), reset them to .1
if (parts.length > 2) {
    for (let i = 2; i < parts.length; i++) {
        newBaseNumberString += ".1";
    }
}
log(`New base number string for next branch topic: ${newBaseNumberString}`);

// 4. Get User Input for the Rest of the New Filename
const userInputFileNamePart = await tp.system.prompt(`新分支主题编号: ${newBaseNumberString}. 请输入文件名其余部分:`, "");
if (userInputFileNamePart === null || userInputFileNamePart === undefined || userInputFileNamePart.trim() === "") {
    log("User cancelled or provided empty filename part for new branch topic.");
    notice("操作已取消：未输入新分支主题文件名。");
    return;
}
const cleanedUserInputPart = userInputFileNamePart.trim().replace(/\.md$/i, "");
log(`Cleaned user input part for new branch topic: ${cleanedUserInputPart}`);

// 5. Build and Check New Filename for Conflicts
let targetNewFileNameWithoutExt = `${newBaseNumberString} ${cleanedUserInputPart}`;
let newFileCounter = 0;
let finalNewFileName = targetNewFileNameWithoutExt + ".md";
let finalNewFilePath = joinPath(parentFolder.path, finalNewFileName);
const originalTargetBaseForNewFile = targetNewFileNameWithoutExt;

log(`Initial target for new branch topic file: ${finalNewFilePath}`);
while (await app.vault.adapter.exists(finalNewFilePath)) {
    newFileCounter++;
    finalNewFileName = `${originalTargetBaseForNewFile}-${newFileCounter}.md`;
    finalNewFilePath = joinPath(parentFolder.path, finalNewFileName);
    log(`Conflict found for new branch topic. Trying new name: ${finalNewFilePath}`);
}
log(`Final non-conflicting path for new branch topic file: ${finalNewFilePath}`);

// 6. Create and Open the New File using content from "原子笔记.md" and add/update category
try {
    const templateFilePath = "06-Templates/原子笔记.md"; // 请再次确认此路径!
    let templateContent = "";
    try {
        templateContent = await app.vault.adapter.read(templateFilePath);
        log(`Successfully read template file for new branch topic: ${templateFilePath}`);

        const fixedCategory = "思考"; // 自动设置为“思考”
        log(`Attempting to set category to: ${fixedCategory} for new branch topic`);

        const frontmatterRegex = /^---\s*\n([\s\S]*?)\n---\s*\n/;
        const fmMatch = templateContent.match(frontmatterRegex);

        if (fmMatch) {
            let fmContent = fmMatch[1];
            const lines = fmContent.split('\n');
            let categoryExists = false;

            for (let i = 0; i < lines.length; i++) {
                if (lines[i].trim().startsWith("category:")) {
                    lines[i] = `category: ${fixedCategory}`;
                    categoryExists = true;
                    log(`Updated existing category line to: category: ${fixedCategory}`);
                    break;
                }
            }

            if (!categoryExists) {
                let insertAtIndex = lines.length; 
                const commonPrecedingFields = ["modified:", "created:", "tags:", "aliases:", "title:", "id:"];
                let lastKnownFieldIndex = -1;

                for (let i = lines.length - 1; i >= 0; i--) {
                    for (const field of commonPrecedingFields) {
                        if (lines[i].trim().startsWith(field)) {
                            lastKnownFieldIndex = i;
                            break;
                        }
                    }
                    if (lastKnownFieldIndex !== -1) break;
                }
                
                if (lastKnownFieldIndex !== -1) {
                    insertAtIndex = lastKnownFieldIndex + 1;
                } else if (lines.length > 0 && lines[0].trim() !== "") { 
                    insertAtIndex = 1; 
                } else {
                    insertAtIndex = 0; 
                }
                
                if (lines.length === 1 && lines[0].trim() === '') {
                    lines[0] = `category: ${fixedCategory}`;
                } else {
                    lines.splice(insertAtIndex, 0, `category: ${fixedCategory}`);
                }
                log(`Added new category line: category: ${fixedCategory}`);
            }
            
            fmContent = lines.join('\n');
            templateContent = templateContent.replace(fmMatch[1], fmContent);
            log(`Frontmatter processed for category for new branch topic.`);

        } else {
            notice("警告：模板文件 \"${templateFilePath}\" 似乎没有标准的 YAML frontmatter。将尝试创建新的 Frontmatter。");
            log("Warning: Standard YAML frontmatter not found in template. Attempting to create new frontmatter for new branch topic.");
            templateContent = `---
category: ${fixedCategory}
---

${templateContent}`;
        }

    } catch (templateError) {
        notice(`读取模板文件 "${templateFilePath}" 失败: ${templateError.message}。将使用空内容。`);
        log(`Error reading template file for new branch topic: ${templateError}`);
        templateContent = `# ${targetNewFileNameWithoutExt}\n\n错误：无法加载模板 "${templateFilePath}"。\n${templateError.message}`;
    }

    const createdFile = await app.vault.create(finalNewFilePath, templateContent);
    log(`Successfully created new branch topic file: ${createdFile.path}`);
    
    let leaf = app.workspace.getLeaf('tab');
    await leaf.openFile(createdFile);
    notice(`新分支主题文件已创建并打开: ${finalNewFileName}`);

} catch (e) {
    notice(`创建新分支主题文件失败: ${e.message}`);
    log(`Error creating new branch topic file: ${e}`);
}

// --- Script End ---
%>
